# Germany demands Bulgaria to become puppet
country_event = {
	id = bulgaria.6
	title = bulgaria.6.t
	desc = bulgaria.6.d
	picture = GFX_report_event_generic_sign_treaty2
	
	is_triggered_only = yes
	
	option = { # Accept demands
		name = bulgaria.6.a
		ai_chance = {	
			factor = 75	
			modifier = {
				factor = 25
				is_in_faction_with = GER
			}
		}
	GER = { puppet = BUL }
	}
	
	option = { # Refuse
		name = bulgaria.6.b
		ai_chance = {
			factor = 20
			modifier = {
				factor = 50
			
					OR = {
						any_other_country = {

							has_guaranteed = BUL
						}
						any_other_country = {

							is_in_faction_with = BUL
						}
					}
				
			}
		}
		if = {
			limit = {
				is_in_faction_with = GER
			}
			GER = { remove_from_faction = BUL }
		}
		GER = { country_event = { id = germany.313 hours = 6 } }
		effect_tooltip = {
			GER = {
				create_wargoal = {
					type = puppet_wargoal_focus
					target = BUL		
				}
			}			
		}
	}
